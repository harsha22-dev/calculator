/* Enable the SPI and make the microcontroller as SPI master */ \
SPCR0 = _BV(SPE0) | _BV(MSTR0); \
SPSR0 = _BV(SPI2X0); /* Double the SPI clock frequency */ \
/* Select the serial clock SCK to be (FOSC/4) \

*/ \
SPCR0 &= ~(_BV(SPR00) | _BV(SPR10)); \
\
/* Set SEL pin which is on Porty to high */ \
