94 void SX8723c_auto_offset(uint8_t device)
95 {
96  int16_t SG_value[4] = {0}; // measurement value of the strain gauge sensor
97  int16_t *p_SG_value = SG_value; // pointer to measurement value
98  int32_t temp = 0; // temporary value for average
99  uint8_t offset = 0; // offset step size
100  uint8_t device_number = 0; // device_number = address in EEPROM
101 
102  switch (device)
103  {
104  case device_1:

106  break;
107 
108  case device_2:

111 
112  case device_3:

124 
125  // reset offset
126  sw_TWI_write_single_register(device, REG_AC_CFG4, 0x00);
127  eeprom_sram_data.SX8723c_config[device_number][EEPROM_SX8723C_REGACCFG4] = 0x00;
128  _delay_ms(4); //wait until new offset is set
129 
130  // get value before offset compensation (average)
131  for (uint16_t i = 0; i < SX8723c_DATA_AVERAGE; i++)
132  {
133  SX8723c_send_request(device);
134  _delay_us(600);
135  SX8723c_get_value(device, p_SG_value); // get value
136  temp += SG_value[device_number];
137  SG_value[device_number] = 0;
138  }
139  SG_value[device_number] = temp / SX8723c_DATA_AVERAGE;
140 
141  //*********************** PGA3 offset **********************
142  // check if offset compensation with PGA3 offset is possible
143  // the smallest step size is 5460 [digit] in both direction
144  if ((SG_value[device_number] > TRESHOLD_PGA3_OFFSET)||(SG_value[device_number] < -TRESHOLD_PGA3_OFFSET))
145  {
146  // value is positive and offset must be negative
147  if (SG_value[device_number] > 0)
148  {
149  // do compensation until value < threshold
150  while (SG_value[device_number] > TRESHOLD_PGA3_OFFSET)
151  {
152  // increment offset PGA3
153  sw_TWI_write_single_register(device, REG_AC_CFG4, (0x40 | ++offset) );
154  _delay_ms(4); //wait until new offset is set
155 
156  // get new value after offset compensation (average)
157  temp = 0;
158  SG_value[device_number] = 0;
159  for (uint16_t i = 0; i < SX8723c_DATA_AVERAGE; i++)
160  {
161  SX8723c_send_request(device);
162  _delay_us(600);
163  SX8723c_get_value(device, p_SG_value); //get value
164  temp += SG_value[device_number];
165  SG_value[device_number] = 0;
166  }
167  SG_value[device_number] = temp / SX8723c_DATA_AVERAGE;
168 
169  // maximal offset is 63
170  // if 63 is reached it is not possible to compensate that huge offset
171  // with PGA3 offset
172  if (offset == 63)
173  {
174  return;
175  }
176  }
177  eeprom_sram_data.SX8723c_config[device_number][EEPROM_SX8723C_REGACCFG4] = (0x40 | offset);
178  }
179 
180  // value is negative and offset must be positive
181  else
