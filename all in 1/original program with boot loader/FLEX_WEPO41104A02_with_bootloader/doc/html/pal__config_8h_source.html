424 /*
425  * TRX Initialization
426  */
427 #define TRX_INIT() do { \
428  /* Enable the SPI and make the microcontroller as SPI master */ \
429  SPCR0 = _BV(SPE0) | _BV(MSTR0); \
430  SPSR0 = _BV(SPI2X0); /* Double the SPI clock frequency */ \
431  /* Select the serial clock SCK to be (FOSC/4) \
432  * For Example for ATmega1281, FOSC = 8Mhz \
433  * (Microcontroller operating frequency) \
434  * SPI clock is thus set to (8/4)*2 = 4 Mhz \
435  */ \
436  SPCR0 &= ~(_BV(SPR00) | _BV(SPR10)); \
437  \
438  /* Set SEL pin which is on Porty to high */ \
439  TRX_PORT2 |= _BV(SEL); \
440 } while (0)
